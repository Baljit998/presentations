Techniques

random

            
  function draw(){
    background(200);
    // random returns a value between the two params
    const randX = random(0, width);
    ellipse(randX, height/2, 10, 10);
  }

grid

            
  const gX = 20;
  const gY = 20;
  const cellSizeX = width/gX;
  const cellSizeY = height/gY;

  for(let i = 0; i < gX; i++){
    for(let j = 0; j < gY; j++){
      const dotSize = random(0, 30);
      const randRed = random(100, 255);
      fill(randRed, 100, 100);
      ellipse(i*cellSizeX, j*cellSizeY, dotSize, dotSize);
    }
  }
